快速开始

安装 SDK

npm install react-native-baidu-map

基础使用

import { MapView, Marker } from 'react-native-baidu-map';

<MapView
  center={{ latitude: 39.9042, longitude: 116.4074 }}
  zoom={15}
>
  <Marker
    coordinate={{ latitude: 39.9042, longitude: 116.4074 }}
    title="天安门"
  />
</MapView>
